## 6. 安装脚本设计

### 6.1 Linux 安装脚本（Bash）

```bash
#!/bin/bash
# mysql-install.sh - MySQL 8.x 安装脚本

set -e

# 配置参数（通过环境变量传入）
MYSQL_ROOT_PASSWORD="${MYSQL_ROOT_PASSWORD:-root123}"
MYSQL_DATA_DIR="${MYSQL_DATA_DIR:-/var/lib/mysql}"
MYSQL_CONFIG_DIR="${MYSQL_CONFIG_DIR:-/etc/mysql}"

log() {
    echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1"
}

# 进度上报（通过特殊输出格式）
report_progress() {
    local stage=$1
    local percent=$2
    echo "PROGRESS:${stage}:${percent}"
}

# 1. 检测依赖
check_dependencies() {
    log "检测系统依赖..."
    report_progress "检查依赖" 10
    
    if command -v curl &> /dev/null; then
        log "✓ curl 已安装"
    else
        log "✗ 缺少 curl，安装中..."
        yum install -y curl || apt-get install -y curl
    fi
}

# 2. 下载安装包
download_package() {
    log "下载安装包..."
    report_progress "下载安装包" 20
    
    # 安装包已在 /tmp 目录（通过 SFTP 上传）
    if [ ! -f "/tmp/mysql-installer.tar.gz" ]; then
        log "错误：安装包不存在"
        exit 1
    fi
}

# 3. 解压安装包
extract_package() {
    log "解压安装包..."
    report_progress "解压安装包" 30
    
    tar -xzf /tmp/mysql-installer.tar.gz -C /tmp/
}

# 4. 安装 MySQL
install_mysql() {
    log "安装 MySQL..."
    report_progress "安装 MySQL" 50
    
    # CentOS/RHEL
    if [ -f /etc/redhat-release ]; then
        yum install -y /tmp/mysql-packages/*.rpm
    # Ubuntu/Debian
    elif [ -f /etc/debian_version ]; then
        dpkg -i /tmp/mysql-packages/*.deb
    fi
}

# 5. 配置 MySQL
configure_mysql() {
    log "配置 MySQL..."
    report_progress "配置 MySQL" 70
    
    # 生成配置文件
    cat > ${MYSQL_CONFIG_DIR}/mysql.cnf <<EOF
[mysqld]
datadir=${MYSQL_DATA_DIR}
port=3306
max_connections=500
EOF
}

# 6. 启动服务
start_service() {
    log "启动 MySQL 服务..."
    report_progress "启动服务" 90
    
    systemctl enable mysqld || systemctl enable mysql
    systemctl start mysqld || systemctl start mysql
    
    sleep 3
    
    if systemctl is-active --quiet mysqld || systemctl is-active --quiet mysql; then
        log "✓ MySQL 启动成功"
        report_progress "完成" 100
    else
        log "✗ MySQL 启动失败"
        exit 1
    fi
}

# 主流程
main() {
    check_dependencies
    download_package
    extract_package
    install_mysql
    configure_mysql
    start_service
    
    log "MySQL 安装完成！"
}

main "$@"
```

### 7.2 进度上报机制

```csharp
// 脚本通过特殊格式输出进度
// 格式：PROGRESS:阶段名称:百分比

public async Task ExecuteScriptAsync(RemoteHost host, string scriptPath)
{
    var session = _connectionService.GetSession(host);
    
    // 执行脚本并实时读取输出
    await session.ExecuteAsync($"bash {scriptPath}", async (output) =>
    {
        if (output.StartsWith("PROGRESS:"))
        {
            // 解析进度上报
            var parts = output.Substring("PROGRESS:".Length).Split(':');
            var stage = parts[0];
            var percent = int.Parse(parts[1]);
            
            // 触发进度事件
            _progressReporter.Report(stage, percent);
        }
        else
        {
            // 普通日志输出
            _logService.Write(LogLevel.Info, output);
        }
    });
}
```

```csharp
public class TaskQueue
{
    private readonly SemaphoreSlim _semaphore;
    private readonly Queue<InstallTask> _queue;
    
    public TaskQueue(int maxConcurrent)
    {
        _semaphore = new SemaphoreSlim(maxConcurrent);
        _queue = new Queue<InstallTask>();
    }
    
    public async Task EnqueueAndExecuteAsync(InstallTask task)
    {
        await _semaphore.WaitAsync();
        try
        {
            await ExecuteTaskAsync(task);
        }
        finally
        {
            _semaphore.Release();
        }
    }
}
```

## 10. 错误处理

### 10.1 重试机制

```csharp
public async Task<string> ExecuteWithRetryAsync(
    Func<Task<string>> action, 
    int maxRetries = 3)
{
    for (int i = 0; i < maxRetries; i++)
    {
        try
        {
            return await action();
        }
        catch (ConnectionException ex) when (i < maxRetries - 1)
        {
            _logService.Write(LogLevel.Warning, 
                $"连接失败，{i + 1}秒后重试...");
            await Task.Delay(1000 * (i + 1));
        }
    }
    throw new MaxRetriesExceededException();
}
```
